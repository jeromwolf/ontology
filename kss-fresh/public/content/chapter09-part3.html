메인 서비스 클래스
ReasoningService.java 
@Service
@Slf4j
public class ReasoningService {
    
    private final OWLReasonerFactory reasonerFactory;
    private final CacheService cacheService;
    private final MetricsService metricsService;
    
    @Autowired
    public ReasoningService(
            @Qualifier("elkReasonerFactory") OWLReasonerFactory reasonerFactory,
            CacheService cacheService,
            MetricsService metricsService) {
        this.reasonerFactory = reasonerFactory;
        this.cacheService = cacheService;
        this.metricsService = metricsService;
    }
    
    /**
     * 클래스 계층 구조 추론
     */
    @Cacheable(value = "class-hierarchy", key = "#ontologyId")
    public ClassHierarchyResponse inferClassHierarchy(String ontologyId) {
        Timer.Sample sample = metricsService.startTimer("reasoning.class-hierarchy");
        
        try {
            // 온톨로지 로드
            OWLOntology ontology = loadOntology(ontologyId);
            
            // ELK 추론 엔진 초기화
            OWLReasoner reasoner = reasonerFactory.createReasoner(ontology);
            
            // 클래스 계층 구조 계산
            Node<OWLClass> topNode = reasoner.getTopClassNode();
            Set<Node<OWLClass>> subClasses = reasoner.getSubClasses(
                ontology.getOWLOntologyManager().getOWLDataFactory().getOWLThing(),
                false
            );
            
            // 결과 구성
            ClassHierarchyResponse response = ClassHierarchyResponse.builder()
                .ontologyId(ontologyId)
                .rootClasses(convertToClassNodes(subClasses))
                .inferenceTime(sample.stop(metricsService.getTimer("reasoning.class-hierarchy")))
                .timestamp(Instant.now())
                .build();
            
            log.info("Class hierarchy inference completed for ontology: {}", ontologyId);
            return response;
            
        } catch (Exception e) {
            metricsService.incrementCounter("reasoning.errors", "type", "class-hierarchy");
            log.error("Failed to infer class hierarchy for ontology: {}", ontologyId, e);
            throw new ReasoningException("Class hierarchy inference failed", e);
        }
    }
    
    /**
     * 실시간 인스턴스 분류
     */
    @Async("reasoningExecutor")
    public CompletableFuture<InstanceClassificationResponse> classifyInstanceAsync(
            String ontologyId, 
            String individualId,
            Map<String, Object> properties) {
        
        return CompletableFuture.supplyAsync(() -> {
            Timer.Sample sample = metricsService.startTimer("reasoning.instance-classification");
            
            try {
                // 캐시 확인
                String cacheKey = generateCacheKey(ontologyId, individualId, properties);
                InstanceClassificationResponse cached = cacheService.get(cacheKey, 
                    InstanceClassificationResponse.class);
                
                if (cached != null) {
                    metricsService.incrementCounter("reasoning.cache.hits");
                    return cached;
                }
                
                // 온톨로지 로드 및 추론
                OWLOntology ontology = loadOntology(ontologyId);
                OWLReasoner reasoner = reasonerFactory.createReasoner(ontology);
                
                // 개체 생성 및 속성 추가
                OWLNamedIndividual individual = createIndividualWithProperties(
                    ontology, individualId, properties);
                
                // 타입 추론
                NodeSet<OWLClass> types = reasoner.getTypes(individual, false);
                Set<OWLClass> inferredClasses = types.getFlattened();
                
                InstanceClassificationResponse response = InstanceClassificationResponse.builder()
                    .ontologyId(ontologyId)
                    .individualId(individualId)
                    .inferredClasses(convertToClassInfo(inferredClasses))
                    .confidence(calculateConfidence(inferredClasses))
                    .inferenceTime(sample.stop(metricsService.getTimer("reasoning.instance-classification")))
                    .build();
                
                // 캐시 저장 (TTL: 1시간)
                cacheService.put(cacheKey, response, Duration.ofHours(1));
                metricsService.incrementCounter("reasoning.cache.misses");
                
                return response;
                
            } catch (Exception e) {
                metricsService.incrementCounter("reasoning.errors", "type", "instance-classification");
                throw new CompletionException(e);
            }
        });
    }
    
    /**
     * 일관성 검증
     */
    public ConsistencyCheckResponse checkConsistency(String ontologyId) {
        Timer.Sample sample = metricsService.startTimer("reasoning.consistency-check");
        
        try {
            OWLOntology ontology = loadOntology(ontologyId);
            OWLReasoner reasoner = reasonerFactory.createReasoner(ontology);
            
            boolean isConsistent = reasoner.isConsistent();
            Set<OWLClass> unsatisfiableClasses = Collections.emptySet();
            
            if (!isConsistent) {
                unsatisfiableClasses = reasoner.getUnsatisfiableClasses().getFlattened();
            }
            
            return ConsistencyCheckResponse.builder()
                .ontologyId(ontologyId)
                .isConsistent(isConsistent)
                .unsatisfiableClasses(convertToClassInfo(unsatisfiableClasses))
                .checkTime(sample.stop(metricsService.getTimer("reasoning.consistency-check")))
                .build();
                
        } catch (Exception e) {
            metricsService.incrementCounter("reasoning.errors", "type", "consistency-check");
            throw new ReasoningException("Consistency check failed", e);
        }
    }
    
    private OWLOntology loadOntology(String ontologyId) {
        // 온톨로지 로딩 로직
        // TDB에서 온톨로지 데이터 로드
        return ontologyRepository.findById(ontologyId)
            .orElseThrow(() -> new OntologyNotFoundException(ontologyId));
    }
    
    private double calculateConfidence(Set<OWLClass> classes) {
        // 신뢰도 계산 로직
        return Math.min(0.95, 0.8 + (classes.size() * 0.05));
    }
}

@Data
@Builder
@JsonInclude(JsonInclude.Include.NON_NULL)
public class ClassHierarchyResponse {
    private String ontologyId;
    private List<ClassNode> rootClasses;
    private Duration inferenceTime;
    private Instant timestamp;
    
    @Data
    @Builder
    public static class ClassNode {
        private String className;
        private String label;
        private List<ClassNode> subClasses;
        private int instanceCount;
    }
}

@Data
@Builder
public class InstanceClassificationResponse {
    private String ontologyId;
    private String individualId;
    private List<ClassInfo> inferredClasses;
    private double confidence;
    private Duration inferenceTime;
    
    @Data
    @Builder
    public static class ClassInfo {
        private String className;
        private String label;
        private double probability;
        private List<String> reasons;
    }
}

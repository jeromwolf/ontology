SPARQL 쿼리 미리보기
PREFIX geo: <http://www.opengis.net/ont/geosparql#>
PREFIX news: <http://example.org/news#>

SELECT ?event ?location ?distance
WHERE {
    ?event news:eventLocation ?location .
    ?location geo:hasGeometry ?geom .
    ?geom geo:asWKT ?wkt .
    
    FILTER(geof:distance(?wkt, 
        "POINT(126.978 37.566)"^^geo:wktLiteral) < 50)
}
ORDER BY ?distance

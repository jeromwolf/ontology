📋 NewsArticle
📝 메타데이터
🏷️ headline string
📅 publishedDate dateTime
📰 publisher NewsSource
👥 개체명

🏢 Organization
📍 Location
🎯 Event
RDF/OWL 코드 미리보기
@prefix news: <http://example.org/news#> .
@prefix owl: <http://www.w3.org/2002/07/owl#> .
@prefix rdfs: <http://www.w3.org/2000/01/rdf-schema#> .

news:NewsArticle rdf:type owl:Class ;
    rdfs:label "뉴스 기사"@ko ;
    rdfs:comment "뉴스 미디어에서 발행된 기사"@ko .

news:Person rdf:type owl:Class ;
    rdfs:subClassOf news:NamedEntity ;
    rdfs:label "인물"@ko .

news:headline rdf:type owl:DatatypeProperty ;
    rdfs:domain news:NewsArticle ;
    rdfs:range xsd:string ;
    rdfs:label "제목"@ko .

IBM Watson for Oncology

온톨로지 기반 암 진단 지원 시스템

# 암 진단 온톨로지 예시
@prefix cancer: <http://example.org/oncology#> .

cancer:BreastCancer rdf:type owl:Class ;
    rdfs:subClassOf cancer:Cancer ;
    cancer:hasRiskFactor cancer:GeneticPredisposition,
                        cancer:Age,
                        cancer:FamilyHistory ;
    cancer:hasStage cancer:Stage0,
                    cancer:StageI,
                    cancer:StageII,
                    cancer:StageIII,
                    cancer:StageIV ;
    cancer:hasTreatment cancer:Surgery,
                        cancer:Chemotherapy,
                        cancer:RadiationTherapy .

# 치료 결정 규칙
cancer:TreatmentDecision(?patient, ?treatment) ← 
    cancer:hasCondition(?patient, ?cancer) ∧
    cancer:hasStage(?cancer, ?stage) ∧
    cancer:recommendedFor(?treatment, ?stage) ∧
    cancer:noContraindication(?patient, ?treatment)
도입 효과
  • 진단 정확도 96% 달성

@prefix fibo-fnd: <https://spec.edmcouncil.org/fibo/ontology/FND/> .
@prefix fibo-be: <https://spec.edmcouncil.org/fibo/ontology/BE/> .

fibo-be:LegalEntity rdf:type owl:Class ;
    rdfs:label "법인"@ko ;
    rdfs:subClassOf fibo-fnd:Organization .

fibo-be:Corporation rdfs:subClassOf fibo-be:LegalEntity ;
    rdfs:label "주식회사"@ko ;
    fibo-be:hasJurisdiction fibo-be:Jurisdiction ;
    fibo-be:isListedOn fibo-fbc:Exchange .

# AML (자금세탁방지) 규칙
fibo-reg:SuspiciousTransaction(?trans) ← 
    fibo-fbc:Transaction(?trans) ∧
    fibo-fbc:hasAmount(?trans, ?amount) ∧
    swrlb:greaterThan(?amount, 10000) ∧
    fibo-fbc:hasFrequency(?trans, ?freq) ∧
    swrlb:greaterThan(?freq, 5)

@prefix mfg: <http://example.org/manufacturing#> .
@prefix iot: <http://example.org/iot#> .

# 생산 설비 온톨로지
mfg:ProductionEquipment rdf:type owl:Class ;
    rdfs:label "생산 설비"@ko .

mfg:Robot rdfs:subClassOf mfg:ProductionEquipment ;
    mfg:hasCapability mfg:Welding,
                      mfg:Assembly,
                      mfg:Painting ;
    iot:hasStatus iot:OperationalStatus ;
    iot:hasSensor iot:TemperatureSensor,
                  iot:VibrationSensor .

# 예측 유지보수 규칙
mfg:MaintenanceRequired(?equipment) ← 
    mfg:ProductionEquipment(?equipment) ∧
    iot:vibrationLevel(?equipment, ?vib) ∧
    swrlb:greaterThan(?vib, 10) ∧
    iot:operatingHours(?equipment, ?hours) ∧
    swrlb:greaterThan(?hours, 1000)

@prefix schema: <https://schema.org/> .

gs1:Product rdf:type owl:Class ;
    owl:equivalentClass schema:Product ;
    rdfs:label "상품"@ko .

gs1:hasGTIN rdf:type owl:DatatypeProperty ;
    rdfs:domain gs1:Product ;
    rdfs:range xsd:string ;
    owl:cardinality 1 .

# 상품 추천 규칙
ecom:RecommendedProduct(?product2, ?customer) ←

    gs1:hasCategory(?product1, ?category) ∧
    gs1:hasCategory(?product2, ?category) ∧
    gs1:hasBrand(?product1, ?brand) ∧
    gs1:hasBrand(?product2, ?brand) ∧
    owl:differentFrom(?product1, ?product2)

@prefix edu: <http://example.org/education#> .
@prefix lom: <http://ltsc.ieee.org/xsd/LOM> .

edu:LearningObject rdf:type owl:Class ;
    rdfs:label "학습 객체"@ko .

edu:Course rdfs:subClassOf edu:LearningObject ;
    edu:hasPrerequisite edu:Course ;
    edu:hasLearningOutcome edu:LearningOutcome ;
    edu:hasDifficulty edu:DifficultyLevel ;
    edu:estimatedTime xsd:duration .

# 적응형 학습 경로 추천
edu:RecommendedNext(?course2, ?student) ← 
    edu:Completed(?student, ?course1) ∧
    edu:hasLearningOutcome(?course1, ?outcome1) ∧
    edu:hasPrerequisite(?course2, ?course1) ∧
    edu:hasSkillLevel(?student, ?skill) ∧
    edu:requiresSkillLevel(?course2, ?required) ∧
    swrlb:greaterThanOrEqual(?skill, ?required)

15.6 스마트 시티

도시 인프라 온톨로지

@prefix city: <http://example.org/smartcity#> .
@prefix geo: <http://www.opengis.net/ont/geosparql#> .

city:CityInfrastructure rdf:type owl:Class ;
    rdfs:label "도시 인프라"@ko .

city:TransportationSystem rdfs:subClassOf city:CityInfrastructure ;
    city:hasCapacity xsd:integer ;
    city:currentUtilization xsd:decimal ;
    geo:hasGeometry geo:Geometry .

city:TrafficLight rdfs:subClassOf city:TransportationSystem ;
    city:hasPhase city:Green, city:Yellow, city:Red ;
    city:connectedTo city:TrafficSensor ;
    city:controlledBy city:TrafficManagementSystem .

# 교통 최적화 규칙
city:AdjustSignalTiming(?light, ?newTiming) ← 
    city:TrafficLight(?light) ∧
    city:nearbyTraffic(?light, ?volume) ∧
    swrlb:greaterThan(?volume, 100) ∧
    city:currentPhase(?light, city:Red) ∧
    city:waitTime(?light, ?wait) ∧
    swrlb:greaterThan(?wait, 120)
